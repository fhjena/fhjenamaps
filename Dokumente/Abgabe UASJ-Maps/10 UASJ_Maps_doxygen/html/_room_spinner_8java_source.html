00055     public String getString() {
00056         String shortenedRoom = (String) spRoom.getSelectedItem(); // kompletter angezeigter String des Raums
00057         int x = shortenedRoom.indexOf(" "); // Index des Leerzeichens herausfinden; gibt -1 zurück, wenn kein Leerzeichen vorhanden
00058         if (-1 != x) // wenn Leerzeichen vorhanden
00059             shortenedRoom = shortenedRoom.substring(0, x); // Leerzeichen und alles dahinter abschneiden
00060         return "'%" + spHouse.getSelectedItem() + "." + spFloor.getSelectedItem() + "." + shortenedRoom + ",%'"; // HausID.EtagenID.RaumID
00061     }
